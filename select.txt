#include <arpa/inet.h> 
#include <netinet/in.h> 
#include <signal.h> 
#include <stdio.h> 
#include <stdlib.h> 
#include <string.h> 
#include <sys/socket.h> 
#include <sys/types.h> 
#include <unistd.h> 


int main(int argc, char const *argv[])
{
int fd;
struct timeval tv;
fd_set read;

tv.tv_sec=1;
tv.tv_usec=2000;

FD_ZERO(&read);
FD_SET(fd, &read);

select(1, &read,NULL,NULL,&tv);

if(FD_ISSET(fd, &read))
{
printf("A key was Pressed\n");

}
else
{
printf("Time Out\n");
}
}
